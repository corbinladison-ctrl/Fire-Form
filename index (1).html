const HEADERS = [ "Reference ID", "Timestamp", "Incident Type", "Building", "Specific Location", "Issue Type", "Severity / Status", "NCC Action", "NCC Reminder", "Notes", "Checklist Completed" ]; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { let data; if (e.parameter && e.parameter.data) { data = JSON.parse(decodeURIComponent(e.parameter.data)); } else if (e.postData && e.postData.contents) { data = JSON.parse(e.postData.contents); } else { throw new Error("No data received"); } const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); sheet.getRange(1, 1, 1, HEADERS.length) .setFontWeight("bold").setBackground("#f0a500").setFontColor("#000000"); sheet.setFrozenRows(1); sheet.setColumnWidth(1, 140); sheet.setColumnWidth(2, 160); sheet.setColumnWidth(10, 260); sheet.setColumnWidth(11, 300); } sheet.appendRow([ data.id || "", data.timestamp || "", data.incident_type || "", data.building || "", data.location || "", data.issue_type || "", data.severity || "", data.ncc_action || "", data.reminder || "", data.notes || "", data.checklist || "" ]); return ContentService.createTextOutput(JSON.stringify({ status: "ok", id: data.id })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // Run this manually to verify sheet setup before deploying function testSetup() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); sheet.appendRow(HEADERS); sheet.getRange(1,1,1,HEADERS.length).setFontWeight("bold").setBackground("#f0a500").setFontColor("#000000"); sheet.setFrozenRows(1); Logger.log("Sheet created."); } else { Logger.log("Sheet already exists."); } }
